105bool AirQualityMonitor::isInternetAvailable()
106{
107 // Tworzenie dedykowanego managera sieci do tego testu
108 QNetworkAccessManager manager;
109 QNetworkRequest request(QUrl(kApiBaseUrl + "station/findAll"));
110
111 // Ustawienie limitu czasu dla żądania
112 request.setTransferTimeout(5000); // 5 sekund timeout
113
114 QNetworkReply* reply = manager.get(request);
115
116 QEventLoop loop;
117 QObject::connect(reply, &QNetworkReply::finished, &loop, &QEventLoop::quit);
118
119 // Również wyjdź z pętli jeśli żądanie przekroczy limit czasu
120 QTimer timer;
121 timer.setSingleShot(true);
122 QObject::connect(&timer, &QTimer::timeout, &loop, &QEventLoop::quit);
123 timer.start(5000); // 5 sekund timeout
124
125 loop.exec(); // Czekaj na odpowiedź lub timeout
126
127 bool success = (reply->error() == QNetworkReply::NoError);
128 reply->deleteLater();
129
130 return success;

213void AirQualityMonitor::updateSensorsFile(const QJsonArray& newSensors)
214{
215 QJsonArray allSensors;
216
217 // Wczytaj istniejące dane, jeśli plik istnieje
218 QFile readFile(QDir::currentPath() + "/sensors.json");
219 if (readFile.exists() && readFile.open(QIODevice::ReadOnly)) {
220 QByteArray data = readFile.readAll();
221 QJsonDocument doc = QJsonDocument::fromJson(data);
222 if (doc.isArray()) {
223 allSensors = doc.array();
224 }
225 readFile.close();
226 }
227
228 // Usuń stare dane dla tej stacji, jeśli istnieją
229 int stationId = -1;
230 if (!newSensors.isEmpty()) {
231 stationId = newSensors.at(0).toObject().value("stationId").toInt();
232 }
233
234 if (stationId != -1) {
235 for (int i = allSensors.size() - 1; i >= 0; i--) {
236 QJsonObject obj = allSensors.at(i).toObject();
237 if (obj.value("stationId").toInt() == stationId) {
238 allSensors.removeAt(i);
239 }
240 }
241 }
242
243 // Dodaj nowe dane
244 for (const QJsonValue& value : newSensors) {
245 allSensors.append(value);
246 }
247
248 // Zapisz zaktualizowane dane do pliku
249 saveSensorsToFile(allSensors);

402void AirQualityMonitor::onMeasurementsDownloaded()
403{
404 QNetworkReply* reply = qobject_cast<QNetworkReply*>(sender());
405 if (!reply) return;
406
407 int sensorId = reply->property("sensorId").toInt();
408
409 if (reply->error() != QNetworkReply::NoError) {
410 qDebug() << "Błąd sieci przy pobieraniu pomiarów:" << reply->errorString();
411 QMessageBox::warning(this, "Błąd pobierania",
412 QString("Nie udało się pobrać danych z serwera: %1\n"
413 "Sprawdzam dane lokalne...")
414 .arg(reply->errorString()), QMessageBox::Ok);
415
416 // Próba załadowania danych offline
417 onMeasurementsLoadedFromFile(sensorId);
418 reply->deleteLater();
419 return;
420 }
421
422 QByteArray data = reply->readAll();
423 QJsonDocument doc = QJsonDocument::fromJson(data);
424
425 if (!doc.isObject()) {
426 qDebug() << "Nieprawidłowy format danych z API";
427 QMessageBox::warning(this, "Błąd formatu",
428 "Dane pobrane z serwera mają nieprawidłowy format.\n"
429 "Sprawdzam dane lokalne...", QMessageBox::Ok);
430
431 // Próba załadowania danych offline
432 onMeasurementsLoadedFromFile(sensorId);
433 reply->deleteLater();
434 return;
435 }
436
437 QJsonObject root = doc.object();
438 QJsonArray values = root.value("values").toArray();
439
440 // Sprawdź czy otrzymaliśmy jakiekolwiek poprawne dane
441 bool hasValidData = false;
442 for (const QJsonValue& val : values) {
443 QJsonObject obj = val.toObject();
444 if (obj.contains("value") && !obj.value("value").isNull()) {
445 hasValidData = true;
446 break;
447 }
448 }
449
450 if (!hasValidData) {
451 QMessageBox::warning(this, "Brak danych",
452 "Serwer nie zwrócił żadnych ważnych danych pomiarowych.\n"
453 "Sprawdzam dane lokalne...", QMessageBox::Ok);
454
455 // Próba załadowania danych offline
456 onMeasurementsLoadedFromFile(sensorId);
457 reply->deleteLater();
458 return;
459 }
460
461 // Otrzymano poprawne dane, zapisz je
462 updateMeasurementsFile(sensorId, values);
463
464 // Aktualizuj wyświetlanie
465 updateMeasurementsList(values);
466
467 // Zaktualizuj również wyświetlanie pomiarów za pomocą wykresu
468 displayMeasurementData(values);
469
470 reply->deleteLater();
471
472 QMessageBox::information(this, "Sukces",
473 "Pomyślnie pobrano najnowsze dane z serwera.", QMessageBox::Ok);
474}

483void AirQualityMonitor::onMeasurementsLoadedFromFile(int sensorId)
484{
485 QJsonArray allMeasurements = loadMeasurementsFromFile();
486 QJsonArray sensorMeasurements;
487 QString lastUpdated = "Nieznany";
488
489 // Znajdź pomiary dla wymaganego sensora
490 for (const QJsonValue& value : allMeasurements) {
491 QJsonObject obj = value.toObject();
492 if (obj.value("id").toInt() == sensorId) {
493 sensorMeasurements = obj.value("values").toArray();
494 // Pobierz timestamp ostatniej aktualizacji jeśli dostępny
495 if (obj.contains("lastUpdated")) {
496 lastUpdated = obj.value("lastUpdated").toString();
497 }
498 break;
499 }
500 }
501
502 if (sensorMeasurements.isEmpty()) {
503 // Spróbuj pobrać dane online jeśli to możliwe
504 if (!isInternetAvailable()) {
505 QMessageBox::warning(this, "Brak danych",
506 "Nie znaleziono zapisanych danych pomiarowych dla tego sensora.\n"
507 "Dodatkowo brak połączenia z internetem - nie można pobrać nowych danych.",
508 QMessageBox::Ok);
509 return;
510 }
511
512 // Jeśli mamy połączenie, pobierz nowe dane
513 QUrl url(QString("https://api.gios.gov.pl/pjp-api/rest/data/getData/%1").arg(sensorId));
514 QNetworkRequest request(url);
515 QNetworkReply* reply = networkManager->get(request);
516 reply->setProperty("sensorId", sensorId); // Zapisz ID sensora jako właściwość
517 connect(reply, &QNetworkReply::finished, this, &AirQualityMonitor::onMeasurementsDownloaded);
518 }
519 else {
520 // Mamy dane offline, używamy ich
521 updateMeasurementsList(sensorMeasurements);
522 displayMeasurementData(sensorMeasurements);
523
524 // Poinformuj użytkownika, że używamy danych z pamięci podręcznej i kiedy były aktualizowane

531 .arg(displayTime),

533

536 QMessageBox::StandardButton reply = QMessageBox::question(this, "Połączenie dostępne",

618 for (QListWidgetItem* item : validItems) {

621

640void AirQualityMonitor::updateMeasurementsFile(int sensorId, const QJsonArray& newValues)
641{
642 // Utwórz timestamp dla danych
643 QDateTime currentTime = QDateTime::currentDateTime();
644 QString timestamp = currentTime.toString(Qt::ISODate);
645
646 QFile file(QDir::currentPath() + "/measurements.json");
647 QJsonArray allMeasurements;
648
649 // Odczytaj istniejące dane jeśli plik istnieje
650 if (file.exists() && file.open(QIODevice::ReadOnly)) {
651 QByteArray data = file.readAll();
652 QJsonDocument doc = QJsonDocument::fromJson(data);
653 if (doc.isArray()) allMeasurements = doc.array();
654 file.close();
655 }
656
657 // Sprawdź czy mamy już dane dla tego sensora
658 bool sensorFound = false;
659 for (int i = 0; i < allMeasurements.size(); i++) {
660 QJsonObject obj = allMeasurements[i].toObject();
661 if (obj.value("id").toInt() == sensorId) {
662 // Aktualizuj istniejące dane sensora
663 sensorFound = true;
664
665 // Dodaj timestamp wskazujący kiedy dane były ostatnio aktualizowane
666 obj["lastUpdated"] = timestamp;
667 obj["values"] = newValues;
668 allMeasurements[i] = obj;
669 break;
670 }
671 }
672
673 // Jeśli sensor nie istnieje w naszym magazynie, dodaj go
674 if (!sensorFound) {
675 QJsonObject newEntry;
676 newEntry["id"] = sensorId;
677 newEntry["values"] = newValues;
678 newEntry["lastUpdated"] = timestamp;
679 allMeasurements.append(newEntry);
680 }
681
682 // Zapisz wszystkie pomiary z powrotem do pliku
683 if (file.open(QIODevice::WriteOnly)) {
684 file.write(QJsonDocument(allMeasurements).toJson());
685 file.close();
686 QMessageBox::information(this, "Informacja", "Dane pomiarowe zostały zapisane do pliku", QMessageBox::Ok);
687 }
688 else {
689 QMessageBox::warning(this, "Błąd", "Nie udało się zapisać danych do pliku. Sprawdź uprawnienia.", QMessageBox::Ok);
690 qDebug() << "Błąd zapisu do measurements.json";
691 }

701void AirQualityMonitor::displayMeasurementData(const QJsonArray& values)
702{
703 if (values.isEmpty())
704 return;
705
706 lastMeasurements = values;
707
708 QDateTime minDate = QDateTime::currentDateTime();
709 QDateTime maxDate = QDateTime::fromSecsSinceEpoch(0);
710
711 for (const QJsonValue& value : values) {
712 QJsonObject obj = value.toObject();
713 QDateTime dt = QDateTime::fromString(obj.value("date").toString(), Qt::ISODate);
714 if (dt.isValid()) {
715 if (dt < minDate) minDate = dt;
716 if (dt > maxDate) maxDate = dt;
717 }
718 }

732void AirQualityMonitor::updateMeasurementDisplay()
733{
734 ui.stationParameterListWidget->clear();
735
736 QLayoutItem* item;
737 while ((item = ui.verticalLayout->takeAt(0)) != nullptr) {
738 if (item->widget())
739 delete item->widget();
740 delete item;
741 }
742
743
744 if (lastMeasurements.isEmpty()) {
745 ui.minValueLabel->setText("Wartość minimalna\nBrak danych");
746 ui.maxValueLabel->setText("Wartość maksymalna\nBrak danych");
747 ui.avgValueLabel->setText("Wartość średnia\nBrak danych");
748 ui.trendLabel->setText("Trend wykresu\nBrak danych");
749 return;
750 }
751
752 QLineSeries* series = new QLineSeries();
753 QList<double> selectedValues;
754
755 for (const QJsonValue& value : lastMeasurements) {
756 QJsonObject obj = value.toObject();
757 QDateTime dt = QDateTime::fromString(obj.value("date").toString(), Qt::ISODate);
758 if (dt.isValid() && obj.contains("value") && !obj.value("value").isNull()) {
759 if (dt >= ui.startDateEdit->dateTime() && dt <= ui.endDateEdit->dateTime()) {
760 double val = obj.value("value").toDouble();
761 selectedValues.append(val);
762 series->append(dt.toMSecsSinceEpoch(), val);
763 ui.stationParameterListWidget->addItem(dt.toString("yyyy-MM-dd HH:mm") + ": " + QString::number(val));
764 }
765 }
766 }
767
768 if (selectedValues.isEmpty()) {
769 ui.minValueLabel->setText("Wartość minimalna\nBrak danych");
770 ui.maxValueLabel->setText("Wartość maksymalna\nBrak danych");
771 ui.avgValueLabel->setText("Wartość średnia\nBrak danych");
772 ui.trendLabel->setText("Trend wykresu\nBrak danych");
773 }
774 else {
775 double min = *std::min_element(selectedValues.begin(), selectedValues.end());
776 double max = *std::max_element(selectedValues.begin(), selectedValues.end());
777 double avg = std::accumulate(selectedValues.begin(), selectedValues.end(), 0.0) / selectedValues.size();
778 QString trend;
779
780 double sumFirst = 0, sumLast = 0;
781 int size = selectedValues.size();
782
783 // Liczymy średnią z pierwszej połowy
784 for (int i = 0; i < size / 2; ++i) {
785 sumFirst += selectedValues[i];
786 }
787
788 // Liczymy średnią z drugiej połowy
789 for (int i = size / 2; i < size; ++i) {
790 sumLast += selectedValues[i];
791 }
792
793 double avgFirst = sumFirst / (size / 2);
794 double avgLast = sumLast / (size - size / 2); // druga połowa może być o 1 większa
795
796 // Określamy trend
797 if (avgLast > avgFirst) {
798 trend = "Rosnący";
799 }
800 else if (avgLast < avgFirst) {
801 trend = "Malejący";
802 }
803 else {
804 trend = "Stabilny";
805 }
806
807
808 // Styl + wyśrodkowanie
809 QString labelStyle = "font-size: 18px; font-weight: bold; color: #00FFC6;";
810 QString trendStyle = "font-size: 18px; font-weight: bold; color: #00FFC6;";
811
812 ui.minValueLabel->setStyleSheet(labelStyle);
813 ui.maxValueLabel->setStyleSheet(labelStyle);
814 ui.avgValueLabel->setStyleSheet(labelStyle);
815 ui.trendLabel->setStyleSheet(trendStyle);
816
817 ui.minValueLabel->setAlignment(Qt::AlignCenter);
818 ui.maxValueLabel->setAlignment(Qt::AlignCenter);
819 ui.avgValueLabel->setAlignment(Qt::AlignCenter);
820 ui.trendLabel->setAlignment(Qt::AlignCenter);
821
822 // Ustawiamy tekst z podpisem i wartością
823 ui.minValueLabel->setText(QString("Wartość minimalna\n%1").arg(QString::number(min, 'f', 2)));
824 ui.maxValueLabel->setText(QString("Wartość maksymalna\n%1").arg(QString::number(max, 'f', 2)));
825 ui.avgValueLabel->setText(QString("Wartość średnia\n%1").arg(QString::number(avg, 'f', 2)));
826 ui.trendLabel->setText(QString("Trend wykresu\n%1").arg(trend));
827 }
828
829 // Wykres
830 QChart* chart = new QChart();
831 chart->legend()->hide();
832 chart->addSeries(series);
833 chart->setTitle("Pomiary");
834
835 QDateTimeAxis* axisX = new QDateTimeAxis;
836 axisX->setFormat("dd-MM HH:mm");
837 axisX->setTitleText("Czas");
838 axisX->setLabelsAngle(-45);
839 chart->addAxis(axisX, Qt::AlignBottom);
840 series->attachAxis(axisX);
841
842 QValueAxis* axisY = new QValueAxis;
843 axisY->setTitleText("Wartość");
844 chart->addAxis(axisY, Qt::AlignLeft);
845 series->attachAxis(axisY);
846
847 chart->setBackgroundBrush(QBrush(QColor("#121212")));
848 chart->setTitleBrush(QBrush(Qt::white));
849 axisX->setLinePenColor(Qt::white);
850 axisX->setLabelsBrush(Qt::white);
851 axisY->setLinePenColor(Qt::white);
852 axisY->setLabelsBrush(Qt::white);
853 axisX->setGridLineColor(QColor("#555555"));
854 axisY->setGridLineColor(QColor("#555555"));
855 series->setColor(QColor("#00c3ff"));
856
857 QChartView* chartView = new QChartView(chart);
858 chartView->setRenderHint(QPainter::Antialiasing);
859
860 ui.verticalLayout->addWidget(chartView);

1048void AirQualityMonitor::findStationsInRadius(double centerLat, double centerLon, double radiusKm)
1049{
1050 QVector<QJsonObject> stationsInRadius;
1051
1052 for (const QJsonValue& value : cachedStations) {
1053 QJsonObject obj = value.toObject();
1054 double lat = obj.value("gegrLat").toString().toDouble();
1055 double lon = obj.value("gegrLon").toString().toDouble();
1056
1057 double distance = haversineDistance(centerLat, centerLon, lat, lon);
1058 if (distance <= radiusKm) {
1059 stationsInRadius.append(obj);
1060 }
1061 }
1062
1063 updateMapWithStations(stationsInRadius);
1064}

1077double AirQualityMonitor::haversineDistance(double lat1, double lon1, double lat2, double lon2)
1078{
1079 double dLat = qDegreesToRadians(lat2 - lat1);
1080 double dLon = qDegreesToRadians(lon2 - lon1);
1081
1082 double a = sin(dLat / 2) * sin(dLat / 2) +
1083 cos(qDegreesToRadians(lat1)) * cos(qDegreesToRadians(lat2)) *
1084 sin(dLon / 2) * sin(dLon / 2);
1085
1086 double c = 2 * atan2(sqrt(a), sqrt(1 - a));
1087
1088 return kEarthRadiusKm * c;
1089}
